from a import HandStrength, calculate_hand_strenght, calculate_hand_values, total_winnings

def test_calculate_hand_strenght():
    assert calculate_hand_strenght('AAAAA') == HandStrength.FIVE_OAK.value
    assert calculate_hand_strenght('AA8AA') == HandStrength.FOUR_OAK.value
    assert calculate_hand_strenght('23332') == HandStrength.FULL_HOUSE.value
    assert calculate_hand_strenght('TTT98') == HandStrength.THREE_OAK.value
    assert calculate_hand_strenght('23432') == HandStrength.PAIR.value
    assert calculate_hand_strenght('A23A4') == HandStrength.PAIR.value
    assert calculate_hand_strenght('23456') == HandStrength.HIGH_CARD.value

def test_calculate_hand_values():
    assert calculate_hand_values('AKQJT') == [14, 13, 12, 11, 10]
    assert calculate_hand_values('23456') == [2, 3, 4, 5, 6]

def test_total_winnings():
    hands = ['32T3K 765', 'T55J5 684', 'KK677 28', 'KTJJT 220', 'QQQJA 483']
    assert total_winnings(hands) == 6440



if __name__ == '__main__':
    test_calculate_hand_strenght()
    test_calculate_hand_values()
    test_total_winnings()
